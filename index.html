1. Estructuras de Control

If...Else

function verificarNumero(num) {
    if (num > 0) {
        return "El número es positivo.";
    } else if (num < 0) {
        return "El número es negativo.";
    } else {
        return "El número es cero.";
    }
}

Switch

function diaDeLaSemana(dia) {
    switch (dia) {
        case 1: return "Lunes";
        case 2: return "Martes";
        case 3: return "Miércoles";
        case 4: return "Jueves";
        case 5: return "Viernes";
        case 6: return "Sábado";
        case 7: return "Domingo";
        default: return "Día no válido.";
    }
}

2. Manejo de Errores

Try/Catch/Throw

function dividir(a, b) {
    try {
        if (b === 0) throw new Error("No se puede dividir por cero.");
        return a / b;
    } catch (error) {
        return error.message;
    }
}

Objetos Error

function lanzarError() {
    throw new Error("Este es un error lanzado intencionadamente.");
}

3. Bucles e Iteración

For

for (let i = 0; i < 5; i++) {
    console.log(i);
}

While

let i = 0;
while (i < 5) {
    console.log(i);
    i++;
}

Do...While

let j = 0;
do {
    console.log(j);
    j++;
} while (j < 5);

5. Expresiones y Operadores

Asignación y Comparación

let a = 5; // Asignación
let b = 10;
let esIgual = (a == b); // Comparación
console.log(esIgual); // false

Operadores Aritméticos

let suma = a + b; // Suma
let resta = b - a; // Resta
let multiplicacion = a * b; // Multiplicación
let division = b / a; // División
console.log(suma, resta, multiplicacion, division); // 15, 5, 50, 2

Operadores Lógicos

let c = true;
let d = false;
let resultado = c && d; // AND
console.log(resultado); // false

Números y Fechas

Objeto Number

let numero = 42;
console.log(Number.isInteger(numero)); // true

Objeto Math

console.log(Math.sqrt(16)); // 4
console.log(Math.random()); // Número aleatorio entre 0 y 1

Objeto Date

let fecha = new Date();
console.log(fecha); // Fecha actual

Formateo de Texto

Cadenas Literales

let nombre = "Juan";
let saludo = \`Hola, \${nombre}!\`; // Cadenas literales
console.log(saludo); // Hola, Juan!

Objeto String

let texto = "Hola Mundo";
console.log(texto.toUpperCase()); // HOLA MUNDO

Colecciones Indexadas

Arreglos

let frutas = ["manzana", "banana", "cereza"];
console.log(frutas[1]); // banana

Arreglos Tipados

let arregloTipado = new Uint8Array(3); // Arreglo tipado
arregloTipado[0] = 10;
console.log(arregloTipado); // Uint8Array(3) [10, 0, 0]
